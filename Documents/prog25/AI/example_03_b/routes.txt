# criar uma pessoa
curl -X POST -H "Content-Type: application/json" -d '{"nome":"Maria", "idade":28, "cidade":"Belo Horizonte"}' http://localhost:5000/pessoas

# criar o carro da pessoa
curl -X POST -H "Content-Type: application/json" -d '{"fabricante":"Volkswagen", "modelo":"Gol", "ano":2020, "dono_id":1}' http://localhost:5000/carros

# listar carro
curl http://localhost:5000/carros
curl http://localhost:5000/carros/1

# atualizar o carro
curl -X PUT -H "Content-Type: application/json" -d '{"fabricante":"Ford", "modelo":"Fiesta", "ano":2022, "dono_id":1}' http://localhost:5000/carros/1

# deletar o carro
curl -X DELETE http://localhost:5000/carros/1
